const SHEETS = {
  leads: 'DASHBOARD_LEADS',
  adSpend: 'AD_SPEND',
  firstRaw: 'FIRST_DB_RAW',
  secondRaw: 'SECOND_DB_RAW',
  mapping: 'CHANNEL_MAPPING',
  users: 'ADMIN_USERS',
  kpiTargets: 'KPI_TARGETS'
};

function doGet(e) {
  const type = e.parameter.type;
  if (type === 'authStatus') return authStatus(e.parameter.token || '');

  const user = sessionUser(e.parameter.token || '');
  if (!user) return json({ error: 'unauthorized' });
  const menu = String(e.parameter.menu || '');
  if (!canAccessMenu(user, menu) || !canReadData(type, menu)) return json({ error: 'forbidden' });

  if (type === 'leads') return getSheetData(SHEETS.leads);
  if (type === 'adSpend') return getSheetData(SHEETS.adSpend);
  if (type === 'firstRaw') return getSheetData(SHEETS.firstRaw);
  if (type === 'secondRaw') return getSheetData(SHEETS.secondRaw);
  if (type === 'mapping') return getSheetData(SHEETS.mapping);
  if (type === 'kpiTargets') return getKpiTargetData();
  if (type === 'dataUpdatedAt') return getDataUpdatedAt();
  return json({ error: 'Invalid type' });
}

function getDataUpdatedAt() {
  const book = SpreadsheetApp.getActive();
  const updatedAt = DriveApp.getFileById(book.getId()).getLastUpdated();
  return json({ updatedAt: updatedAt.toISOString() });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const type = body.type;
    const rows = body.rows || [];

    if (type === 'authSetup') return authSetup(body);
    if (type === 'authLogin') return authLogin(body);
    if (type === 'authLogout') return authLogout(body.token || '');

    const user = sessionUser(body.token || '');
    if (!user) return json({ error: 'unauthorized' });
    if (type === 'authListUsers') return authListUsers(user);
    if (type === 'authCreateUser') return authCreateUser(user, body);
    if (type === 'authUpdateUser') return authUpdateUser(user, body);

    const menu = resolveWriteMenu(type, String(body.menu || ''));
    if (!canAccessMenu(user, menu) || !canWriteData(type, menu)) return json({ error: 'forbidden' });

    if (type === 'leads') return appendRowsUnique(SHEETS.leads, rows, 'leads');
    if (type === 'adSpend') return appendRowsUnique(SHEETS.adSpend, rows, 'adSpend');
    if (type === 'adSpendReplace') return replaceAdSpendRows(rows);
    if (type === 'updateAdSpend') return updateAdSpendRow(body);
    if (type === 'deleteAdSpend') return deleteAdSpendRow(body);
    if (type === 'firstRaw') return appendRowsUnique(SHEETS.firstRaw, rows, 'firstRaw');
    if (type === 'secondRaw') return appendRowsUnique(SHEETS.secondRaw, rows, 'secondRaw');
    if (type === 'leadCorrections') return updateLeadCorrections(body);
    if (type === 'updateLead') return updateLeadRow(body);
    if (type === 'saveKpiTarget') return saveKpiTarget(body, user);

    return json({ error: 'Invalid type' });
  } catch (err) {
    return json({ error: String(err) });
  }
}

function getSheetData(sheetName) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(sheetName);
  if (!sheet) return json({ error: 'Sheet not found: ' + sheetName });
  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json([]);
  const headers = values.shift();

  const result = values
    .filter(row => row.some(cell => cell !== ''))
    .map(row => {
      const obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });

  return json(result);
}

function ensureKpiTargetSheet() {
  const book = SpreadsheetApp.getActive();
  let sheet = book.getSheetByName(SHEETS.kpiTargets);
  if (!sheet) sheet = book.insertSheet(SHEETS.kpiTargets);
  if (sheet.getLastRow() === 0) {
    sheet.getRange(1, 1, 1, 5).setValues([['month', 'minDaily', 'stretchDaily', 'updatedBy', 'updatedAt']]);
    sheet.setFrozenRows(1);
  }
  if (sheet.getLastRow() === 1) {
    sheet.getRange(2, 1, 1, 5).setValues([['2026-07', 40, 60, 'system', new Date().toISOString()]]);
  }
  return sheet;
}

function getKpiTargetData() {
  const sheet = ensureKpiTargetSheet();
  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json([]);
  const headers = values.shift();
  return json(values.filter(row => row.some(Boolean)).map(row => {
    const item = {};
    headers.forEach((header, index) => item[header] = row[index]);
    return item;
  }));
}

function saveKpiTarget(body, user) {
  if (!requireMaster(user)) return json({ error: 'forbidden' });
  const target = body.target || {};
  const month = String(target.month || '').slice(0, 7);
  const minDaily = Math.max(0, Math.round(Number(target.minDaily) || 0));
  const stretchDaily = Math.max(minDaily, Math.round(Number(target.stretchDaily) || 0));
  if (!/^\d{4}-\d{2}$/.test(month)) return json({ error: 'invalid target month' });
  if (minDaily <= 0 || stretchDaily <= 0) return json({ error: 'invalid target value' });

  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const sheet = ensureKpiTargetSheet();
    const values = sheet.getDataRange().getDisplayValues();
    let rowNumber = -1;
    for (let row = 1; row < values.length; row++) {
      if (String(values[row][0] || '').slice(0, 7) === month) {
        rowNumber = row + 1;
        break;
      }
    }
    const output = [[month, minDaily, stretchDaily, String(user.name || user.id || ''), new Date().toISOString()]];
    if (rowNumber > 0) sheet.getRange(rowNumber, 1, 1, 5).setValues(output);
    else sheet.getRange(sheet.getLastRow() + 1, 1, 1, 5).setValues(output);
    return json({ success: true, target: {
      month: month,
      minDaily: minDaily,
      stretchDaily: stretchDaily,
      updatedBy: String(user.name || user.id || '')
    }});
  } finally {
    lock.releaseLock();
  }
}

function appendRowsUnique(sheetName, rows, type) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(sheetName);
  if (!sheet) return json({ error: 'Sheet not found: ' + sheetName });
  if (!rows || rows.length === 0) return json({ success: true, count: 0, skipped: 0 });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];

  // 업로드 데이터에 새 컬럼이 있으면 시트 헤더를 자동 확장
  const incomingKeys = [];
  rows.forEach(row => Object.keys(row).forEach(k => { if (incomingKeys.indexOf(k) === -1) incomingKeys.push(k); }));
  incomingKeys.forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });

  if (headers.length === 0) return json({ error: 'No headers found' });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const existing = buildExistingKeySet(sheet, headers, type);
  const toAppend = [];
  const seenIncoming = {};

  rows.forEach(row => {
    const key = makeUniqueKey(row, type);
    if (!key) return;
    if (existing[key] || seenIncoming[key]) return;
    seenIncoming[key] = true;
    toAppend.push(row);
  });

  if (toAppend.length === 0) return json({ success: true, count: 0, skipped: rows.length });

  const values = toAppend.map(row => headers.map(h => row[h] === undefined || row[h] === null ? '' : row[h]));
  sheet.getRange(sheet.getLastRow() + 1, 1, values.length, headers.length).setValues(values);
  return json({ success: true, count: values.length, skipped: rows.length - values.length });
}

function replaceAdSpendRows(rows) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });
  if (!rows || rows.length === 0) return json({ success: true, count: 0, deleted: 0 });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  const incomingKeys = [];
  rows.forEach(row => Object.keys(row).forEach(k => { if (incomingKeys.indexOf(k) === -1) incomingKeys.push(k); }));
  incomingKeys.forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  if (headers.length === 0) return json({ error: 'No headers found' });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const dateIdx = headers.indexOf('date');
  const channelIdx = headers.indexOf('channel');
  if (dateIdx < 0 || channelIdx < 0) return json({ error: 'AD_SPEND date/channel column not found' });

  const targets = {};
  rows.forEach(row => {
    const date = String(row.date || '').slice(0, 10);
    const channel = String(row.channel || '').toLowerCase();
    if (date && channel) targets[date + '_' + channel] = true;
  });

  const values = sheet.getDataRange().getDisplayValues();
  let deleted = 0;
  for (let r = values.length - 1; r >= 1; r--) {
    const date = String(values[r][dateIdx] || '').slice(0, 10);
    const channel = String(values[r][channelIdx] || '').toLowerCase();
    if (targets[date + '_' + channel]) {
      sheet.deleteRow(r + 1);
      deleted++;
    }
  }

  const rowsToAppend = rows.filter(function(row) {
    return Number(String(row.amount || row.cost || row['광고비'] || row['비용'] || 0).replace(/[^0-9]/g, '')) > 0;
  });
  if (rowsToAppend.length === 0) {
    return json({ success: true, count: 0, skipped: rows.length, deleted: deleted, replaced: true });
  }

  const appendResult = JSON.parse(appendRowsUnique(SHEETS.adSpend, rowsToAppend, 'adSpend').getContent());
  appendResult.deleted = deleted;
  appendResult.replaced = true;
  return json(appendResult);
}

function adSpendRowKey(row) {
  const date = String(row.date || row['날짜'] || '').slice(0, 10);
  const channel = String(row.channel || row['매체'] || '').toLowerCase();
  const subChannel = String(row.subChannel || row['상세매체'] || '').toLowerCase();
  const campaign = String(row.campaign || row['캠페인'] || '').toLowerCase();
  return date + '_' + channel + '_' + subChannel + '_' + campaign;
}

function updateAdSpendRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });

  const original = body.original || {};
  const patch = body.patch || {};
  const originalKey = adSpendRowKey(original);
  if (!originalKey) return json({ error: 'original ad spend key required' });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  Object.keys(patch).forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const values = sheet.getDataRange().getDisplayValues();
  let targetRow = -1;
  for (let r = 1; r < values.length; r++) {
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === originalKey) {
      targetRow = r + 1;
      break;
    }
  }
  if (targetRow < 0) return json({ error: 'matching ad spend not found' });

  const next = {};
  headers.forEach((h, i) => next[h] = values[targetRow - 1][i] || '');
  Object.keys(patch).forEach(k => next[k] = patch[k]);
  const nextKey = adSpendRowKey(next);

  for (let r = 1; r < values.length; r++) {
    if (r + 1 === targetRow) continue;
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === nextKey) return json({ error: 'duplicate ad spend' });
  }

  Object.keys(patch).forEach(k => {
    sheet.getRange(targetRow, headers.indexOf(k) + 1).setValue(patch[k]);
  });
  return json({ success: true, updated: 1 });
}

function deleteAdSpendRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });

  const originalKey = adSpendRowKey(body.original || {});
  if (!originalKey) return json({ error: 'original ad spend key required' });
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getDisplayValues()[0];
  const values = sheet.getDataRange().getDisplayValues();

  for (let r = 1; r < values.length; r++) {
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === originalKey) {
      sheet.deleteRow(r + 1);
      return json({ success: true, deleted: 1 });
    }
  }
  return json({ error: 'matching ad spend not found' });
}


function baseLeadStage(v) {
  const stage = String(v || '').toLowerCase();
  if (stage === 'first_reentry') return 'first';
  if (stage === 'second_reentry') return 'second';
  return stage;
}

function dateTimeKey(v) {
  const text = String(v || '').replace('T', ' ').replace('Z', '').trim();
  const match = text.match(/^(\d{4}-\d{2}-\d{2})\s+(\d{1,2}):(\d{1,2})/);
  return match ? match[1] + ' ' + ('0' + match[2]).slice(-2) + ':' + ('0' + match[3]).slice(-2) : text.slice(0, 16);
}

function updateLeadRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.leads);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.leads });

  const phone = normalizePhone(body.phone || '');
  const stage = baseLeadStage(body.stage || '');
  const patch = body.patch || {};
  const targetDate = String(body.date || '').slice(0, 10);
  const targetRegisteredAt = dateTimeKey(body.registeredAt || '');
  if (!phone || !stage) return json({ error: 'phone/stage required' });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  Object.keys(patch).forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  if (headers.indexOf('changeHistory') === -1) headers.push('changeHistory');
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json({ error: 'No leads data' });
  const h = values[0];
  const phoneIdx = h.indexOf('phone');
  const stageIdx = h.indexOf('stage') >= 0 ? h.indexOf('stage') : h.indexOf('dbTier');
  const dateIdx = h.indexOf('date');
  const registeredAtIdx = h.indexOf('registeredAt');
  if (phoneIdx < 0 || stageIdx < 0) return json({ error: 'phone/stage column not found' });

  let matchedRow = -1;
  let fallbackRow = -1;
  for (let r = 1; r < values.length; r++) {
    const rowPhone = normalizePhone(values[r][phoneIdx]);
    const rowStage = baseLeadStage(values[r][stageIdx]);
    const rowDate = dateIdx >= 0 ? String(values[r][dateIdx] || '').slice(0, 10) : '';
    if (rowPhone !== phone || rowStage !== stage || (targetDate && rowDate !== targetDate)) continue;
    fallbackRow = r;
    const rowRegisteredAt = registeredAtIdx >= 0 ? dateTimeKey(values[r][registeredAtIdx]) : '';
    if (!targetRegisteredAt || (rowRegisteredAt && rowRegisteredAt === targetRegisteredAt)) {
      matchedRow = r;
      break;
    }
  }
  if (matchedRow < 0) matchedRow = fallbackRow;
  if (matchedRow < 0) return json({ success: false, updated: 0, error: 'matching lead not found' });

  const now = new Date().toISOString();
  const operator = patch.operator || patch.updatedBy || '';
  const historyIdx = headers.indexOf('changeHistory');
  const oldHistory = historyIdx >= 0 ? String(values[matchedRow][historyIdx] || '') : '';
  const changedKeys = Object.keys(patch).filter(k => k !== 'updatedAt' && k !== 'updatedBy' && patch[k] !== undefined && patch[k] !== null).join(', ');
  const action = String(patch.status || '').toLowerCase() === 'deleted' ? '삭제' : '수정';
  const historyLine = '[' + now.replace('T',' ').slice(0,16) + '] ' + (operator || '작업자미상') + ' ' + action + ': ' + changedKeys;
  patch.updatedAt = now;
  patch.changeHistory = oldHistory ? oldHistory + '\n' + historyLine : historyLine;

  Object.keys(patch).forEach(k => {
    const col = h.indexOf(k) >= 0 ? h.indexOf(k) : headers.indexOf(k);
    sheet.getRange(matchedRow + 1, col + 1).setValue(patch[k]);
  });
  return json({ success: true, updated: 1 });
}

function updateLeadCorrections(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.leads);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.leads });

  const items = body.rows || [];
  if (!items.length) return json({ success: true, updated: 0, requested: 0 });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  if (headers.indexOf('changeHistory') === -1) headers.push('changeHistory');
  items.forEach(item => {
    const patch = item.patch || {};
    Object.keys(patch).forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json({ error: 'No leads data' });
  values[0] = headers.slice();
  for (let r = 0; r < values.length; r++) {
    while (values[r].length < headers.length) values[r].push('');
  }

  const phoneIdx = headers.indexOf('phone');
  const stageIdx = headers.indexOf('stage') >= 0 ? headers.indexOf('stage') : headers.indexOf('dbTier');
  const dateIdx = headers.indexOf('date');
  const registeredAtIdx = headers.indexOf('registeredAt');
  const consultingIdx = headers.indexOf('consultingNumber');
  const historyIdx = headers.indexOf('changeHistory');
  if (phoneIdx < 0 || stageIdx < 0) return json({ error: 'phone/stage column not found' });

  let updated = 0;
  items.forEach(item => {
    const phone = normalizePhone(item.phone || '');
    const stage = baseLeadStage(item.stage || '');
    const targetDate = String(item.matchDate || item.date || '').slice(0, 10);
    const targetRegisteredAt = dateTimeKey(item.registeredAt || '');
    const consultingNumber = String(item.consultingNumber || '').trim();
    if (!phone || !stage) return;

    let matchedRow = -1;
    let fallbackRow = -1;
    for (let r = 1; r < values.length; r++) {
      if (consultingNumber && consultingIdx >= 0 && String(values[r][consultingIdx] || '').trim() === consultingNumber) {
        matchedRow = r;
        break;
      }
      const rowPhone = normalizePhone(values[r][phoneIdx]);
      const rowStage = baseLeadStage(values[r][stageIdx]);
      const rowDate = dateIdx >= 0 ? String(values[r][dateIdx] || '').slice(0, 10) : '';
      if (rowPhone !== phone || rowStage !== stage || (targetDate && rowDate !== targetDate)) continue;
      fallbackRow = r;
      const rowRegisteredAt = registeredAtIdx >= 0 ? dateTimeKey(values[r][registeredAtIdx]) : '';
      if (!targetRegisteredAt || !rowRegisteredAt || rowRegisteredAt === targetRegisteredAt) {
        matchedRow = r;
        break;
      }
    }
    if (matchedRow < 0) matchedRow = fallbackRow;
    if (matchedRow < 0) return;

    const patch = item.patch || {};
    const now = new Date().toISOString();
    const operator = patch.operator || patch.updatedBy || '';
    const changedKeys = Object.keys(patch).filter(k => k !== 'updatedAt' && k !== 'updatedBy' && patch[k] !== undefined && patch[k] !== null).join(', ');
    const oldHistory = historyIdx >= 0 ? String(values[matchedRow][historyIdx] || '') : '';
    const historyLine = '[' + now.replace('T', ' ').slice(0, 16) + '] ' + (operator || 'system') + ' update: ' + changedKeys;
    patch.updatedAt = now;
    patch.changeHistory = oldHistory ? oldHistory + '\n' + historyLine : historyLine;

    Object.keys(patch).forEach(k => {
      const col = headers.indexOf(k);
      if (col >= 0) values[matchedRow][col] = patch[k] == null ? '' : patch[k];
    });
    updated++;
  });

  sheet.getRange(1, 1, values.length, headers.length).setValues(values);
  return json({ success: true, updated: updated, requested: items.length });
}

function buildExistingKeySet(sheet, headers, type) {
  const set = {};
  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return set;
  values.shift();
  values.forEach(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    const key = makeUniqueKey(obj, type);
    if (key) set[key] = true;
  });
  return set;
}

function normalizePhone(v) {
  let phone = String(v || '').replace(/[^0-9]/g, '');
  if (phone.length === 10 && phone.indexOf('10') === 0) phone = '0' + phone;
  return phone;
}

function makeUniqueKey(row, type) {
  if (type === 'leads') {
    const phone = normalizePhone(row.phone || row._parsed_phone || row['연락처'] || row['휴대폰번호'] || row['휴대폰 번호']);
    const stage = String(row.stage || row.dbTier || row._parsed_stage || '').toLowerCase();
    const date = String(row.date || row._parsed_date || row['등록일시'] || row['등록 일시'] || row['접수일시'] || '').slice(0, 10);
    return phone && stage && date ? phone + '_' + stage + '_' + date : (phone && stage ? phone + '_' + stage : '');
  }

  if (type === 'firstRaw' || type === 'secondRaw') {
    const phone = normalizePhone(row._parsed_phone || row.phone || row['연락처'] || row['휴대폰번호'] || row['휴대폰 번호']);
    const stage = String(row._parsed_stage || row.stage || row.dbTier || (type === 'secondRaw' ? 'second' : 'first')).toLowerCase();
    const date = String(row._parsed_date || row.date || row['등록일시'] || row['등록 일시'] || row['접수일시'] || '').slice(0, 10);
    return phone && stage ? type + '_' + phone + '_' + stage + '_' + date : '';
  }

  if (type === 'adSpend') {
    const date = String(row.date || row['날짜'] || '').slice(0, 10);
    const channel = String(row.channel || row['매체'] || '').toLowerCase();
    const subChannel = String(row.subChannel || row['상세매체'] || '').toLowerCase();
    const campaign = String(row.campaign || row['캠페인'] || '').toLowerCase();
    return date && channel ? date + '_' + channel + '_' + subChannel + '_' + campaign : '';
  }

  return '';
}

const USER_HEADERS = ['id', 'name', 'passwordHash', 'salt', 'role', 'permissions', 'active', 'createdAt', 'updatedAt'];
const SESSION_TTL_SECONDS = 21600;
const SALES_SUPPORT_MENUS = ['/db-manage', '/upload-db', '/region'];
const VALID_MENUS = [
  '/dashboard', '/channels', '/kpi', '/funnel', '/region', '/db-manage',
  '/upload-db', '/upload-spend', '/manage-spend', '/admin-users'
];

function ensureUsersSheet() {
  const book = SpreadsheetApp.getActive();
  let sheet = book.getSheetByName(SHEETS.users);
  if (!sheet) sheet = book.insertSheet(SHEETS.users);
  if (sheet.getLastRow() === 0) {
    sheet.getRange(1, 1, 1, USER_HEADERS.length).setValues([USER_HEADERS]);
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function getUsers() {
  const sheet = ensureUsersSheet();
  if (sheet.getLastRow() <= 1) return [];
  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, USER_HEADERS.length).getDisplayValues();
  return values.filter(row => String(row[0] || '').trim()).map((row, index) => {
    const user = { rowNumber: index + 2 };
    USER_HEADERS.forEach((header, column) => user[header] = row[column]);
    user.active = String(user.active).toLowerCase() !== 'false';
    user.permissions = parsePermissions(user.permissions);
    return user;
  });
}

function findUser(id) {
  const key = String(id || '').trim().toLowerCase();
  if (!key) return null;
  const users = getUsers();
  for (let i = 0; i < users.length; i++) {
    if (String(users[i].id || '').trim().toLowerCase() === key) return users[i];
  }
  return null;
}

function parsePermissions(value) {
  if (Array.isArray(value)) return value.map(String).filter(Boolean);
  return String(value || '').split(/[|,]/).map(value => value.trim()).filter(Boolean);
}

function normalizedPermissions(role, permissions) {
  if (role === 'master') return ['*'];
  if (role === 'sales_support') return SALES_SUPPORT_MENUS.slice();
  const requested = parsePermissions(permissions);
  return VALID_MENUS.filter(menu => requested.indexOf(menu) >= 0);
}

function publicUser(user) {
  return {
    id: String(user.id || ''),
    name: String(user.name || ''),
    role: String(user.role || 'custom'),
    permissions: parsePermissions(user.permissions),
    active: user.active !== false && String(user.active).toLowerCase() !== 'false',
    createdAt: String(user.createdAt || ''),
    updatedAt: String(user.updatedAt || '')
  };
}

function hashPassword(password, salt) {
  const bytes = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    String(salt) + ':' + String(password),
    Utilities.Charset.UTF_8
  );
  return Utilities.base64EncodeWebSafe(bytes);
}

function createSession(userId) {
  const token = Utilities.getUuid() + Utilities.getUuid().replace(/-/g, '');
  CacheService.getScriptCache().put('admin_session_' + token, String(userId), SESSION_TTL_SECONDS);
  return token;
}

function sessionUser(token) {
  if (!token) return null;
  const userId = CacheService.getScriptCache().get('admin_session_' + String(token));
  if (!userId) return null;
  const user = findUser(userId);
  if (!user || !publicUser(user).active) {
    CacheService.getScriptCache().remove('admin_session_' + String(token));
    return null;
  }
  return user;
}

function authStatus(token) {
  const users = getUsers();
  if (users.length === 0) return json({ setupRequired: true });
  const user = sessionUser(token);
  return json({ setupRequired: false, user: user ? publicUser(user) : null });
}

function authSetup(body) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    if (getUsers().length > 0) return json({ error: 'setup already completed' });
    const id = String(body.id || '').trim();
    const name = String(body.name || '').trim() || id;
    const password = String(body.password || '');
    if (!id) return json({ error: 'id required' });
    if (password.length < 8) return json({ error: 'password must be at least 8 characters' });

    const now = new Date().toISOString();
    const salt = Utilities.getUuid();
    const user = {
      id: id,
      name: name,
      passwordHash: hashPassword(password, salt),
      salt: salt,
      role: 'master',
      permissions: ['*'],
      active: true,
      createdAt: now,
      updatedAt: now
    };
    appendUser(user);
    return json({ success: true, token: createSession(id), user: publicUser(user) });
  } finally {
    lock.releaseLock();
  }
}

function authLogin(body) {
  const user = findUser(body.id);
  if (!user || !publicUser(user).active) return json({ error: 'invalid credentials' });
  const actual = hashPassword(String(body.password || ''), user.salt);
  if (actual !== String(user.passwordHash || '')) return json({ error: 'invalid credentials' });
  return json({ success: true, token: createSession(user.id), user: publicUser(user) });
}

function authLogout(token) {
  if (token) CacheService.getScriptCache().remove('admin_session_' + String(token));
  return json({ success: true });
}

function appendUser(user) {
  const sheet = ensureUsersSheet();
  const row = USER_HEADERS.map(header => {
    if (header === 'permissions') return parsePermissions(user[header]).join('|');
    return user[header] === undefined || user[header] === null ? '' : user[header];
  });
  sheet.getRange(sheet.getLastRow() + 1, 1, 1, USER_HEADERS.length).setValues([row]);
}

function requireMaster(user) {
  return user && String(user.role) === 'master' && publicUser(user).active;
}

function authListUsers(actor) {
  if (!requireMaster(actor)) return json({ error: 'forbidden' });
  return json({ users: getUsers().map(publicUser) });
}

function authCreateUser(actor, body) {
  if (!requireMaster(actor)) return json({ error: 'forbidden' });
  const id = String(body.id || '').trim();
  const name = String(body.name || '').trim() || id;
  const password = String(body.password || '');
  const role = ['master', 'sales_support', 'custom'].indexOf(String(body.role)) >= 0 ? String(body.role) : 'custom';
  if (!id) return json({ error: 'id required' });
  if (findUser(id)) return json({ error: 'id already exists' });
  if (password.length < 8) return json({ error: 'password must be at least 8 characters' });

  const now = new Date().toISOString();
  const salt = Utilities.getUuid();
  const user = {
    id: id,
    name: name,
    passwordHash: hashPassword(password, salt),
    salt: salt,
    role: role,
    permissions: normalizedPermissions(role, body.permissions),
    active: true,
    createdAt: now,
    updatedAt: now
  };
  appendUser(user);
  return json({ success: true, user: publicUser(user) });
}

function authUpdateUser(actor, body) {
  if (!requireMaster(actor)) return json({ error: 'forbidden' });
  const user = findUser(body.id);
  if (!user) return json({ error: 'user not found' });

  const isSelf = String(actor.id).toLowerCase() === String(user.id).toLowerCase();
  const role = body.role === undefined ? String(user.role) :
    (['master', 'sales_support', 'custom'].indexOf(String(body.role)) >= 0 ? String(body.role) : 'custom');
  const active = body.active === undefined ? publicUser(user).active : Boolean(body.active);
  if (isSelf && (role !== 'master' || !active)) return json({ error: 'cannot remove your own master access' });

  user.name = body.name === undefined ? user.name : (String(body.name || '').trim() || user.id);
  user.role = role;
  user.permissions = normalizedPermissions(role, body.permissions === undefined ? user.permissions : body.permissions);
  user.active = active;
  user.updatedAt = new Date().toISOString();
  if (String(body.password || '')) {
    if (String(body.password).length < 8) return json({ error: 'password must be at least 8 characters' });
    user.salt = Utilities.getUuid();
    user.passwordHash = hashPassword(String(body.password), user.salt);
  }

  const sheet = ensureUsersSheet();
  const row = USER_HEADERS.map(header => {
    if (header === 'permissions') return parsePermissions(user[header]).join('|');
    return user[header] === undefined || user[header] === null ? '' : user[header];
  });
  sheet.getRange(user.rowNumber, 1, 1, USER_HEADERS.length).setValues([row]);
  return json({ success: true, user: publicUser(user) });
}

function canAccessMenu(user, menu) {
  if (!user || VALID_MENUS.indexOf(menu) < 0) return false;
  if (String(user.role) === 'master') return true;
  const permissions = parsePermissions(user.permissions);
  return permissions.indexOf('*') >= 0 || permissions.indexOf(menu) >= 0;
}

  const rules = {
    leads: ['/dashboard', '/channels', '/kpi', '/funnel', '/region', '/db-manage', '/upload-db'],

function resolveWriteMenu(type, menu) {
  if (type === 'firstRaw' || type === 'secondRaw' || type === 'leadCorrections') return '/upload-db';
  if (type === 'leads') {
    if (menu === '/db-manage' || menu === '/upload-db') return menu;
    return '/upload-db';
  }
  if (type === 'adSpend' || type === 'adSpendReplace') return '/upload-spend';
  if (type === 'updateAdSpend' || type === 'deleteAdSpend') return '/manage-spend';
  if (type === 'saveKpiTarget') return '/kpi';
  return VALID_MENUS.indexOf(menu) >= 0 ? menu : '';
}
