const SHEETS = {
  leads: 'DASHBOARD_LEADS',
  adSpend: 'AD_SPEND',
  firstRaw: 'FIRST_DB_RAW',
  secondRaw: 'SECOND_DB_RAW',
  mapping: 'CHANNEL_MAPPING'
};

function doGet(e) {
  const type = e.parameter.type;
  if (type === 'leads') return getSheetData(SHEETS.leads);
  if (type === 'adSpend') return getSheetData(SHEETS.adSpend);
  if (type === 'firstRaw') return getSheetData(SHEETS.firstRaw);
  if (type === 'secondRaw') return getSheetData(SHEETS.secondRaw);
  if (type === 'mapping') return getSheetData(SHEETS.mapping);
  return json({ error: 'Invalid type' });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const type = body.type;
    const rows = body.rows || [];

    if (type === 'leads') return appendRowsUnique(SHEETS.leads, rows, 'leads');
    if (type === 'adSpend') return appendRowsUnique(SHEETS.adSpend, rows, 'adSpend');
    if (type === 'adSpendReplace') return replaceAdSpendRows(rows);
    if (type === 'updateAdSpend') return updateAdSpendRow(body);
    if (type === 'deleteAdSpend') return deleteAdSpendRow(body);
    if (type === 'firstRaw') return appendRowsUnique(SHEETS.firstRaw, rows, 'firstRaw');
    if (type === 'secondRaw') return appendRowsUnique(SHEETS.secondRaw, rows, 'secondRaw');
    if (type === 'updateLead') return updateLeadRow(body);

    return json({ error: 'Invalid type' });
  } catch (err) {
    return json({ error: String(err) });
  }
}

function getSheetData(sheetName) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(sheetName);
  if (!sheet) return json({ error: 'Sheet not found: ' + sheetName });
  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json([]);
  const headers = values.shift();

  const result = values
    .filter(row => row.some(cell => cell !== ''))
    .map(row => {
      const obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });

  return json(result);
}

function appendRowsUnique(sheetName, rows, type) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(sheetName);
  if (!sheet) return json({ error: 'Sheet not found: ' + sheetName });
  if (!rows || rows.length === 0) return json({ success: true, count: 0, skipped: 0 });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];

  // 업로드 데이터에 새 컬럼이 있으면 시트 헤더를 자동 확장
  const incomingKeys = [];
  rows.forEach(row => Object.keys(row).forEach(k => { if (incomingKeys.indexOf(k) === -1) incomingKeys.push(k); }));
  incomingKeys.forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });

  if (headers.length === 0) return json({ error: 'No headers found' });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const existing = buildExistingKeySet(sheet, headers, type);
  const toAppend = [];
  const seenIncoming = {};

  rows.forEach(row => {
    const key = makeUniqueKey(row, type);
    if (!key) return;
    if (existing[key] || seenIncoming[key]) return;
    seenIncoming[key] = true;
    toAppend.push(row);
  });

  if (toAppend.length === 0) return json({ success: true, count: 0, skipped: rows.length });

  const values = toAppend.map(row => headers.map(h => row[h] === undefined || row[h] === null ? '' : row[h]));
  sheet.getRange(sheet.getLastRow() + 1, 1, values.length, headers.length).setValues(values);
  return json({ success: true, count: values.length, skipped: rows.length - values.length });
}

function replaceAdSpendRows(rows) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });
  if (!rows || rows.length === 0) return json({ success: true, count: 0, deleted: 0 });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  const incomingKeys = [];
  rows.forEach(row => Object.keys(row).forEach(k => { if (incomingKeys.indexOf(k) === -1) incomingKeys.push(k); }));
  incomingKeys.forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  if (headers.length === 0) return json({ error: 'No headers found' });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const dateIdx = headers.indexOf('date');
  const channelIdx = headers.indexOf('channel');
  if (dateIdx < 0 || channelIdx < 0) return json({ error: 'AD_SPEND date/channel column not found' });

  const targets = {};
  rows.forEach(row => {
    const date = String(row.date || '').slice(0, 10);
    const channel = String(row.channel || '').toLowerCase();
    if (date && channel) targets[date + '_' + channel] = true;
  });

  const values = sheet.getDataRange().getDisplayValues();
  let deleted = 0;
  for (let r = values.length - 1; r >= 1; r--) {
    const date = String(values[r][dateIdx] || '').slice(0, 10);
    const channel = String(values[r][channelIdx] || '').toLowerCase();
    if (targets[date + '_' + channel]) {
      sheet.deleteRow(r + 1);
      deleted++;
    }
  }

  const appendResult = JSON.parse(appendRowsUnique(SHEETS.adSpend, rows, 'adSpend').getContent());
  appendResult.deleted = deleted;
  appendResult.replaced = true;
  return json(appendResult);
}

function adSpendRowKey(row) {
  const date = String(row.date || row['날짜'] || '').slice(0, 10);
  const channel = String(row.channel || row['매체'] || '').toLowerCase();
  const subChannel = String(row.subChannel || row['상세매체'] || '').toLowerCase();
  const campaign = String(row.campaign || row['캠페인'] || '').toLowerCase();
  return date + '_' + channel + '_' + subChannel + '_' + campaign;
}

function updateAdSpendRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });

  const original = body.original || {};
  const patch = body.patch || {};
  const originalKey = adSpendRowKey(original);
  if (!originalKey) return json({ error: 'original ad spend key required' });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  Object.keys(patch).forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const values = sheet.getDataRange().getDisplayValues();
  let targetRow = -1;
  for (let r = 1; r < values.length; r++) {
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === originalKey) {
      targetRow = r + 1;
      break;
    }
  }
  if (targetRow < 0) return json({ error: 'matching ad spend not found' });

  const next = {};
  headers.forEach((h, i) => next[h] = values[targetRow - 1][i] || '');
  Object.keys(patch).forEach(k => next[k] = patch[k]);
  const nextKey = adSpendRowKey(next);

  for (let r = 1; r < values.length; r++) {
    if (r + 1 === targetRow) continue;
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === nextKey) return json({ error: 'duplicate ad spend' });
  }

  Object.keys(patch).forEach(k => {
    sheet.getRange(targetRow, headers.indexOf(k) + 1).setValue(patch[k]);
  });
  return json({ success: true, updated: 1 });
}

function deleteAdSpendRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.adSpend);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.adSpend });

  const originalKey = adSpendRowKey(body.original || {});
  if (!originalKey) return json({ error: 'original ad spend key required' });
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getDisplayValues()[0];
  const values = sheet.getDataRange().getDisplayValues();

  for (let r = 1; r < values.length; r++) {
    const row = {};
    headers.forEach((h, i) => row[h] = values[r][i] || '');
    if (adSpendRowKey(row) === originalKey) {
      sheet.deleteRow(r + 1);
      return json({ success: true, deleted: 1 });
    }
  }
  return json({ error: 'matching ad spend not found' });
}


function baseLeadStage(v) {
  const stage = String(v || '').toLowerCase();
  if (stage === 'first_reentry') return 'first';
  if (stage === 'second_reentry') return 'second';
  return stage;
}

function dateTimeKey(v) {
  const text = String(v || '').replace('T', ' ').replace('Z', '').trim();
  const match = text.match(/^(\d{4}-\d{2}-\d{2})\s+(\d{1,2}):(\d{1,2})/);
  return match ? match[1] + ' ' + ('0' + match[2]).slice(-2) + ':' + ('0' + match[3]).slice(-2) : text.slice(0, 16);
}

function updateLeadRow(body) {
  const sheet = SpreadsheetApp.getActive().getSheetByName(SHEETS.leads);
  if (!sheet) return json({ error: 'Sheet not found: ' + SHEETS.leads });

  const phone = normalizePhone(body.phone || '');
  const stage = baseLeadStage(body.stage || '');
  const patch = body.patch || {};
  const targetDate = String(body.date || '').slice(0, 10);
  const targetRegisteredAt = dateTimeKey(body.registeredAt || '');
  if (!phone || !stage) return json({ error: 'phone/stage required' });

  let lastCol = sheet.getLastColumn();
  let headers = lastCol > 0 ? sheet.getRange(1, 1, 1, lastCol).getDisplayValues()[0].filter(String) : [];
  Object.keys(patch).forEach(k => { if (headers.indexOf(k) === -1) headers.push(k); });
  if (headers.indexOf('changeHistory') === -1) headers.push('changeHistory');
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);

  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return json({ error: 'No leads data' });
  const h = values[0];
  const phoneIdx = h.indexOf('phone');
  const stageIdx = h.indexOf('stage') >= 0 ? h.indexOf('stage') : h.indexOf('dbTier');
  const dateIdx = h.indexOf('date');
  const registeredAtIdx = h.indexOf('registeredAt');
  if (phoneIdx < 0 || stageIdx < 0) return json({ error: 'phone/stage column not found' });

  let matchedRow = -1;
  let fallbackRow = -1;
  for (let r = 1; r < values.length; r++) {
    const rowPhone = normalizePhone(values[r][phoneIdx]);
    const rowStage = baseLeadStage(values[r][stageIdx]);
    const rowDate = dateIdx >= 0 ? String(values[r][dateIdx] || '').slice(0, 10) : '';
    if (rowPhone !== phone || rowStage !== stage || (targetDate && rowDate !== targetDate)) continue;
    fallbackRow = r;
    const rowRegisteredAt = registeredAtIdx >= 0 ? dateTimeKey(values[r][registeredAtIdx]) : '';
    if (!targetRegisteredAt || (rowRegisteredAt && rowRegisteredAt === targetRegisteredAt)) {
      matchedRow = r;
      break;
    }
  }
  if (matchedRow < 0) matchedRow = fallbackRow;
  if (matchedRow < 0) return json({ success: false, updated: 0, error: 'matching lead not found' });

  const now = new Date().toISOString();
  const operator = patch.operator || patch.updatedBy || '';
  const historyIdx = headers.indexOf('changeHistory');
  const oldHistory = historyIdx >= 0 ? String(values[matchedRow][historyIdx] || '') : '';
  const changedKeys = Object.keys(patch).filter(k => k !== 'updatedAt' && k !== 'updatedBy' && patch[k] !== undefined && patch[k] !== null).join(', ');
  const action = String(patch.status || '').toLowerCase() === 'deleted' ? '삭제' : '수정';
  const historyLine = '[' + now.replace('T',' ').slice(0,16) + '] ' + (operator || '작업자미상') + ' ' + action + ': ' + changedKeys;
  patch.updatedAt = now;
  patch.changeHistory = oldHistory ? oldHistory + '\n' + historyLine : historyLine;

  Object.keys(patch).forEach(k => {
    const col = h.indexOf(k) >= 0 ? h.indexOf(k) : headers.indexOf(k);
    sheet.getRange(matchedRow + 1, col + 1).setValue(patch[k]);
  });
  return json({ success: true, updated: 1 });
}

function buildExistingKeySet(sheet, headers, type) {
  const set = {};
  const values = sheet.getDataRange().getDisplayValues();
  if (!values || values.length <= 1) return set;
  values.shift();
  values.forEach(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    const key = makeUniqueKey(obj, type);
    if (key) set[key] = true;
  });
  return set;
}

function normalizePhone(v) {
  let phone = String(v || '').replace(/[^0-9]/g, '');
  if (phone.length === 10 && phone.indexOf('10') === 0) phone = '0' + phone;
  return phone;
}

function makeUniqueKey(row, type) {
  if (type === 'leads') {
    const phone = normalizePhone(row.phone || row._parsed_phone || row['연락처'] || row['휴대폰번호'] || row['휴대폰 번호']);
    const stage = String(row.stage || row.dbTier || row._parsed_stage || '').toLowerCase();
    const date = String(row.date || row._parsed_date || row['등록일시'] || row['등록 일시'] || row['접수일시'] || '').slice(0, 10);
    return phone && stage && date ? phone + '_' + stage + '_' + date : (phone && stage ? phone + '_' + stage : '');
  }

  if (type === 'firstRaw' || type === 'secondRaw') {
    const phone = normalizePhone(row._parsed_phone || row.phone || row['연락처'] || row['휴대폰번호'] || row['휴대폰 번호']);
    const stage = String(row._parsed_stage || row.stage || row.dbTier || (type === 'secondRaw' ? 'second' : 'first')).toLowerCase();
    const date = String(row._parsed_date || row.date || row['등록일시'] || row['등록 일시'] || row['접수일시'] || '').slice(0, 10);
    return phone && stage ? type + '_' + phone + '_' + stage + '_' + date : '';
  }

  if (type === 'adSpend') {
    const date = String(row.date || row['날짜'] || '').slice(0, 10);
    const channel = String(row.channel || row['매체'] || '').toLowerCase();
    const subChannel = String(row.subChannel || row['상세매체'] || '').toLowerCase();
    const campaign = String(row.campaign || row['캠페인'] || '').toLowerCase();
    return date && channel ? date + '_' + channel + '_' + subChannel + '_' + campaign : '';
  }

  return '';
}

function json(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
